10  package br.unb.cic.iris;
11  
12  import static br.unb.cic.iris.i18n.Message.message;
13  
14  import java.util.Scanner;
15  
16  import br.unb.cic.iris.command.console.ConsoleCommandManager;
17  import br.unb.cic.iris.core.SystemFacade;
18  
19  public class MainProgram {
20  
21  	Scanner sc;
22  
23  	public MainProgram() {
24  		sc = new Scanner(System.in);
25  	}
26  
27  	public static void main(String args[]) {
28  		MainProgram m = new MainProgram();
29  		//eager loading/starting the system
30  		SystemFacade.instance();
31  		ConsoleCommandManager.singleton().runCommand("");
32  		
33  		m.mainMenu();
34  		m.readCommand();
35  	}
36  
37  	private void mainMenu() {
38  		System.out.println(message("interpreter"));
39  		System.out.println(message("version"));
40  		System.out.println(message("help"));
41  	}
42  
43  	private void readCommand() {
44  		try {
45  			System.out.print(message("prompt"));
46  			String cmd = sc.nextLine().trim();
47  			ConsoleCommandManager.singleton().runCommand(cmd);
48  		} catch (RuntimeException e) {
49  			System.err.println(message("error") + e.getLocalizedMessage());
50  		}
51  		readCommand();
52  	}
53  
54  }
