25  public final class SystemFacade {
26  	private static final SystemFacade instance = new SystemFacade();
27  
28  	private IEmailClient client;
29  	private EmailProvider provider;
30  
31  	private Status status = NOT_CONNECTED;
32  
33  	private SystemFacade() {
34  		// load config file
35  		Configuration config = new Configuration();
36  		// create a default provider based on properties loaded
37  		provider = new DefaultProvider(config.getProperties());
38  		// register the default provider
39  		ProviderManager.instance().addProvider(provider);
40  		// initiate connection
41  		connect(provider);
42  	}
43  
44  	public static SystemFacade instance() {
45  		return instance;
46  	}
47  
48  	public void connect(EmailProvider provider) {
49  		setStatus(NOT_CONNECTED);
50  		this.provider = provider;
51  		client = new EmailClient(provider);
52  		setStatus(CONNECTED);
53  	}
54  
55  	public void send(EmailMessage message) throws EmailException {
56  		verifyConnection();
57  		client.send(message);
58  	}
59  
60  	public List<IrisFolder> listFolders() throws EmailException {
61  		verifyConnection();
62  		return client.listFolders();
63  	}

80  	private void verifyConnection() {
81  		if (!isConnected()) {
82  			throw new EmailUncheckedException(message("error.not.connected"));
83  		}
84  	}
85  
86  	public boolean isConnected() {
