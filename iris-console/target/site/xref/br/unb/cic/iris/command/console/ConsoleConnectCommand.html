26  	@Override
27  	public void execute() {
28  		Scanner sc = new Scanner(System.in);
29  
30  		String providerStr = read(sc, "command.connect.label.provider");
31  		EmailProvider provider = ProviderManager.instance().getProvider(
32  				providerStr);
33  		System.out.println("Prodiver recovered: " + provider.getName());
34  		// TODO validar provider antes de ler o resto
35  
36  		String username = read(sc, "command.connect.label.username");
37  		String password = read(sc, "command.connect.label.password");
38  
39  		provider.setUsername(username);
40  		provider.setPassword(password);
41  
42  		SystemFacade.instance().connect(provider);
43  	}
44  
45  	private static String read(Scanner sc, String question) {
46  		askQuestion(question);
47  		return sc.nextLine();
48  	}
49  
50  	private static void askQuestion(String message) {

52  	}
